\begin{align*} S_N&=\sum_{n=1}^N\log\Big(1+\frac{1}{n}\Big)\\ &=\overbrace{\log\Big(1+\frac{1}{1}\Big)}^{n=1} +\overbrace{\log\Big(1+\frac{1}{2}\Big)}^{n=2} +\overbrace{\log\Big(1+\frac{1}{3}\Big)}^{n=3} + \cdots\\ \amp\hskip1in +\overbrace{\log\Big(1+\frac{1}{N-1}\Big)}^{n=N-1} +\overbrace{\log\Big(1+\frac{1}{N}\Big)}^{n=N}\\ &=\log(2) + \log\Big(\frac{3}{2}\Big) +\log\Big(\frac{4}{3}\Big) +\cdots+ \log\Big(\frac{N}{N-1}\Big) + \log\Big(\frac{N+1}{N}\Big)\\ &=\log\Big(2\times\frac{3}{2}\times\frac{4}{3}\times \cdots \times \frac{N}{N-1}\times \frac{N+1}{N}\Big)\\ &=\log(N+1) \end{align*}
