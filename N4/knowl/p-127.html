\begin{equation*} \frac {1}{1-x} = \sum_{n=0}^{\infty}x^n ,\;\;|x|\lt 1 \end{equation*}
\begin{equation*} =1+x+x^2+x^3+x^4+\cdots ,\;\; |x|\lt 1 \end{equation*}
