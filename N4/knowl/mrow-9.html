\begin{align*} \diff{}{x} \left\{\frac{1}{1-x}\right\} &= \diff{}{x} \sum_{n=0}^\infty x^n & \text{ $|x| \lt 1$}\\ &= \sum_{n=0}^\infty \diff{}{x} x^n & \text{basta derivar termo a termo}\\ &= \sum_{n=0}^\infty n x^{n-1}\\ \end{align*}
